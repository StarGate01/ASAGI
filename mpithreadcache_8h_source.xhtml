489  bool transfer(long entry, unsigned long blockId, unsigned char *cache, bool &retry)
490  {
491  if (entry < 0) {
492  retry = false;
493  return false;
494  }
495 
496  int rank = entry / rankCacheSize();
497 
498  assert(rank >= 0);
499 
500  std::lock_guard<threads::Mutex> lock(*m_blockMutex);
501 
502  // Send the message to the remote rank
503  mpi::CommThread::commThread.send(m_transfererTag, rank, blockId);
504 
505  int mpiResult; NDBG_UNUSED(mpiResult);
506  MPI_Status status;
507  int done = 0;
508  while (!done) {
509  // Try success full transfer first
510  mpiResult = MPI_Iprobe(rank, m_tagOffset+TRANSFER_TAG, mpiComm().comm(), &done, &status);
511  assert(mpiResult == MPI_SUCCESS);
512 
513  if (!done) {
514  // Try unsuccessfull transfer
515  mpiResult = MPI_Iprobe(rank, m_tagOffset+TRANSFER_FAIL_TAG, mpiComm().comm(),
516  &done, &status);
517  assert(mpiResult == MPI_SUCCESS);
518  }
519  }
520 
521  if (status.MPI_TAG == m_tagOffset+TRANSFER_TAG) {
522  // Success
523  mpiResult = MPI_Recv(cache, m_blockSize, m_mpiType, rank, m_tagOffset+TRANSFER_TAG,
524  mpiComm().comm(), MPI_STATUS_IGNORE);
525  assert(mpiResult == MPI_SUCCESS);
526 
527  retry = false;
528  return true;
529  }
530 
531  // Fail
532  char ack;
533  mpiResult = MPI_Recv(&ack, 1, MPI_CHAR, rank, m_tagOffset+TRANSFER_FAIL_TAG,
534  mpiComm().comm(), MPI_STATUS_IGNORE);
535  assert(mpiResult == MPI_SUCCESS);
536 
537  retry = true;
538  return false;
