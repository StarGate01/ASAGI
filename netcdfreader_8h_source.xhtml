189  void getBlock(void *block,
190  const size_t *offset,
191  const size_t *size)
192  {
193  // Make sure we do not read more than the available data
194  // in each dimension
195  size_t actSize[MAX_DIMENSIONS];
196  for (int i = 0; i < m_dimensions; i++) {
197  if (offset[i] + size[i] > getSize(i))
198  actSize[i] = getSize(i) - offset[i];
199  else
200  actSize[i] = size[i];
201  }
202 
203  // The distance between 2 values in the internal representation
204  ptrdiff_t imap[MAX_DIMENSIONS];
205  imap[m_dimensions-1] = getBasicSize<T>();
206  for (int i = m_dimensions-2; i >= 0; i--)
207  imap[i] = imap[i+1] * size[i+1];
208 
209  std::lock_guard<threads::Mutex> lock(netcdfLock);
210  getVar(offset, actSize, imap,
211  static_cast<T*>(block));
212  }
