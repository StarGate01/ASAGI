77 int asagi::Grid::nodeLocalRank(MPI_Comm comm)
78 {
79  // The main idea for this function is taken from:
80  // https://blogs.fau.de/wittmann/2013/02/mpi-node-local-rank-determination/
81  // http://git.rrze.uni-erlangen.de/gitweb/?p=apsm.git;a=blob;f=MpiNodeRank.cpp;hb=HEAD
82 
83  int mpiResult; NDBG_UNUSED(mpiResult);
84 
85  typedef char procName_t[MPI_MAX_PROCESSOR_NAME+1];
86 
87  // Get the processor name
88  procName_t procName;
89  int procNameLength;
90 
91  MPI_Get_processor_name(procName, &procNameLength);
92  assert(procNameLength <= MPI_MAX_PROCESSOR_NAME);
93 
94  procName[procNameLength] = '\0';
95 
96  // Compute Adler32 hash
97  const uint8_t* buffer = reinterpret_cast<const uint8_t*>(procName);
98  uint32_t s1 = 1;
99  uint32_t s2 = 0;
100 
101  for (int i = 0; i < procNameLength; i++) {
102  s1 = (s1 + buffer[i]) % 65521;
103  s2 = (s2 + s1) % 65521;
104  }
105 
106  uint32_t hash = (s2 << 16) | s1;
107  int rank;
108  mpiResult = MPI_Comm_rank(comm, &rank);
109  assert(mpiResult == MPI_SUCCESS);
110 
111  MPI_Comm nodeComm;
112  mpiResult = MPI_Comm_split(comm, hash, rank, &nodeComm);
113  assert(mpiResult == MPI_SUCCESS);
114 
115  // Gather all proc names of this node to detect Adler32 collisions
116  int nodeSize;
117  mpiResult = MPI_Comm_size(nodeComm, &nodeSize);
118  assert(mpiResult == MPI_SUCCESS);
119 
120  procName_t* procNames = new procName_t[nodeSize];
121 
122  mpiResult = MPI_Allgather(procName, MPI_MAX_PROCESSOR_NAME+1, MPI_CHAR,
123  procNames, MPI_MAX_PROCESSOR_NAME+1, MPI_CHAR, nodeComm);
124  assert(mpiResult == MPI_SUCCESS);
125 
126  // recv contains now an array of hostnames from all MPI ranks of
127  // this communicator. They are sorted ascending by the MPI rank.
128  int nodeRank, realNodeRank = 0;
129  mpiResult = MPI_Comm_rank(comm, &nodeRank);
130  assert(mpiResult == MPI_SUCCESS);
131 
132  for (int i = 0; i < nodeRank; i++) {
133  if (strcmp(procName, procNames[i]) == 0)
134  // Detect false hash collisions
135  realNodeRank++;
136  }
137 
138  mpiResult = MPI_Comm_free(&nodeComm);
139  assert(mpiResult == MPI_SUCCESS);
140 
141  delete [] procNames;
142 
143  return realNodeRank;
144 }
